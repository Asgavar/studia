3. Find occupations (P106) practised by persons with a given name Konstanty.
   Return pairs (occupation, the number of persons practising the occupation),
   Sort the results in the descending order by this number, and limit the number of results to 10.

   #+BEGIN_SRC
    SELECT ?occupation ?occupationLabel (COUNT(?occupation) AS ?count) WHERE {
      ?konstanty wdt:P735 wd:Q20087914.
      ?konstanty wdt:P106 ?occupation.
      SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE]". }
    }
    GROUP BY ?occupation ?occupationLabel
    ORDER BY DESC(?count)
    LIMIT 10
